data_bak = np.zeros(data.shape[:2],np.uint8)
def res(rgb):
    return sum(np.power([255, 247, 0]-rgb,2))
for i in range(len(data)):
    for j in range(len(data[i])):
        if res(data[i][j]) < 20000:
            data_bak[i][j] = 255
